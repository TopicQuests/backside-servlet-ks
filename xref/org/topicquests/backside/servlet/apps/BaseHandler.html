105 	// GetTicket is the workhorse of platform security:
106 	//	We must determine:
107 	//		a) who this user is
108 	//			which really means proving this user is not spoofing some other user
109 	//		b) if this user is authenticated
110 	// Theories behind this include:
111 	//		Combination of 
112 	//			username (which must be unique)
113 	//			caller's IP address
114 	//				issues of what happens on mobile users
115 	//		All of which means that the query string must include enough information
116 	//			to determine who this is.
117 	// CODE as written for processRequest really means this:
118 	//	each app should test to make sure that the verb it gets is a legal verb
119 	//  If not, it should throw an exception
120 	///////////////////////////////////////////////////////////////////

122 	/**
123 	 * The game here is to return two objects: {@link ITicket} and {@link JSONObject}
124 	 * @param request
125 	 * @return
126 	 * @throws ServletException
127 	 */
128 	private IResult processRequest(HttpServletRequest request) throws ServletException {
129 		IResult result = new ResultPojo();
130 		ITicket t = guestCredentials; //default
131 		String pt = getPath(request);
132 		//seeing things like auth/{....}
133 		System.out.println("PR "+pt);
134 		//PR {"verb":"PutTopic","uIP":"173.164.129.250","uName":"jackpark","cargo":{"uName
135 		//":"jackpark","Lang":"en","Label":"My test topic","Details":"Created to test the
136 		//AngularApp code","LiP":"/images/cogwheel.png","SiP":"/images/cogwheel.sm.png","I
137 		//sPvt":"F","SType":"ClassType"}}
138 		if (!pt.startsWith("{")) {
139 			int where = pt.indexOf('/');
140 			if (where > -1)  {
141 				pt = pt.substring(where+1);
142 			}
143 		}
144 		JSONObject jo = jsonFromString(pt);
145 		result.setResultObjectA(jo);
146 		String x = jo.getAsString(ICredentialsMicroformat.USER_IP);
147 		System.out.println("GET IP "+x);
148 		//IP is the user's ip address -- it can be null
149 		String verb = getVerb(jo);
150 		if (!verb.equals(IAuthMicroformat.AUTHENTICATE)) {
151 			//IF THIS IS NOT AN AUTHENTICATE: look up this user
152 			String token = jo.getAsString(ICredentialsMicroformat.SESSION_TOKEN);
153 			// can send in either null, or an empty string
154 			if (token != null && !token.equals("")) {
155 				t = credentialCache.getTicket(token);
156 				if (t == null) {
157 					x = IErrorMessages.TOKEN_NO_USER+token;
158 					//CASE: someone sent in a token, but we don't know who it is: big error
159 					// could be that the same platform logged out before
160 					environment.logError(x, null);
161 					//This can happen if the server goes down, then is
162 					//restarted followed by someone logging out.
163 					//We will not toss an exception anymore.
164 					//throw new ServletException(x);
165 				} 
166 			}
167 			//otherwise, there is no token on this transaction
168 		}
169 		result.setResultObject(t);
170 		return result;
171 	}

177 	public void executeGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
178 		IResult r = processRequest(request);
179 		ITicket t = (ITicket)r.getResultObject();
180 		JSONObject jo = (JSONObject)r.getResultObjectA();
181 		handleGet(request,response, t, jo);
182 		r = null;
183 		t = null;
184 		jo = null;
185 	}

345 		File f = new File(pdfPath);
346 		System.out.println("SENDPDF "+pdfPath);
347 		if (!f.exists())
348 			throw new IOException("NoFILE "+pdfPath);
349 		ServletOutputStream stream = null;
350 	    BufferedInputStream buf = null;
351 	    try 
352 	    {
353 	        stream = response.getOutputStream();
354 	        response.setContentType("application/pdf");
355 	        response.setHeader("Content-Disposition", "inline; filename='" + f.getName() + "'");
356 	        FileInputStream input = new FileInputStream(f);
357 	        response.setContentLength((int) f.length());
358 	        buf = new BufferedInputStream(input);
359 	        int readBytes = 0;
360 	        while ((readBytes = buf.read()) != -1)
361 	            stream.write(readBytes);
362 	    } 
363 	    finally 
364 	    {
365 	        if (stream != null)
366 	            stream.close();
367 	        if (buf != null)
368 	            buf.close();
369 	    }
370 	}

372 	public String getClientIpAddr(HttpServletRequest request) {  
373         String ip = request.getHeader("X-Forwarded-For");  
374         if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {  
375             ip = request.getHeader("Proxy-Client-IP");  
376         }  
377         if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {  
378             ip = request.getHeader("WL-Proxy-Client-IP");  
379         }  
380         if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {  
381             ip = request.getHeader("HTTP_CLIENT_IP");  
382         }  
383         if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {  
384             ip = request.getHeader("HTTP_X_FORWARDED_FOR");  
385         }  
386         if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {  
387             ip = request.getRemoteAddr();  
388         }  
389         return ip;  
390     }

443 	 * Convert <code>jsonString</code> to a {@link JSONObject}
444 	 * @param jsonString

446 	 * @throws ServletException

448 	public JSONObject jsonFromString(String jsonString) throws ServletException {
449 		//environment.logDebug("JSONFROMSTRING "+jsonString);
450 		//NOTE: there are edge conditions:
451 		//  jsonString == ""  can happen
452 		JSONParser p = new JSONParser(JSONParser.MODE_JSON_SIMPLE);
453 		try {
454 			return (JSONObject)p.parse(jsonString);
455 		} catch (Exception e) {
456 			environment.logError(jsonString, e);
457 			throw new ServletException(e);
458 		}
459 	}

472 		//name, email, avatar, homepage, geolocation, role
473 		JSONObject result = newJSONObject();
474 		result.put(IUserMicroformat.USER_EMAIL, t.getProperty(IUserSchema.USER_EMAIL));
475 		result.put(IUserMicroformat.USER_HANDLE, t.getProperty(IUserSchema.USER_NAME));
476 		result.put(IUserMicroformat.USER_ID, t.getUserLocator());
477 		String ava = "";
478 		List<String> avas = t.listAvatars();
479 		if (avas != null && avas.size() > 0)
480 			ava = avas.get(0); //FOR NOW, just show first one
481 		result.put(IUserMicroformat.USER_AVATAR, ava);
482 		result.put(IUserMicroformat.USER_HOMEPAGE, notNullString((String)t.getProperty(IUserSchema.USER_HOMEPAGE)));
483 		result.put(IUserMicroformat.USER_ROLE, t.getProperty(IUserMicroformat.USER_ROLE));
484 		result.put(IUserMicroformat.USER_GEOLOC, t.getProperty(IUserSchema.USER_GEOLOC));
485 		result.put(IUserMicroformat.USER_FULLNAME, t.getProperty(IUserSchema.USER_FULLNAME));
