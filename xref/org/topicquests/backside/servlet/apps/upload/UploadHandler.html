126 	String getCoreQuery(String verb, String userId, String userIP, String sToken) throws ServletException {
127 		try {
128 			JSONObject json = new JSONObject();
129 			json.put(ICredentialsMicroformat.VERB, verb);
130 			json.put(ICredentialsMicroformat.USER_ID, userId);
131 			json.put(ICredentialsMicroformat.SESSION_TOKEN, sToken);
132 			json.put(ICredentialsMicroformat.USER_IP, userIP);
133 			String result = json.toJSONString();
134 			result = URLEncoder.encode(result, "UTF-8");
135 			json = null;
136 			return result;
137 		} catch (Exception e) {
138 			environment.logError(e.getMessage(), e);
139 			throw new ServletException(e);
140 		}
