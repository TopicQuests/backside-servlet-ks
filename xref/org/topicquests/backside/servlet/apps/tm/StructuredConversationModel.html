148 	@Override
149 	public IResult transcludeChildNode(String parentLocator, String contextLocator, String childLocator,
150 			String language, ITicket credentials) {
151 		IResult result = new ResultPojo();
152 		IResult r = topicMap.getNode(parentLocator, credentials);
153 		if (r.hasError()) result.addErrorString(r.getErrorString());
154 		ISubjectProxy parent = (ISubjectProxy)r.getResultObject();
155 		r = topicMap.getNode(childLocator, credentials);
156 		if (r.hasError()) result.addErrorString(r.getErrorString());
157 		ISubjectProxy child = (ISubjectProxy)r.getResultObject();
158 		((IParentChildContainer)child).addParentNode(contextLocator, parent.getSmallImage(), parentLocator, parent.getLabel(language));
159 		((IParentChildContainer)parent).addChildNode(contextLocator, child.getSmallImage(), childLocator, child.getLabel(language), childLocator);
160 		child.doUpdate();
161 		parent.doUpdate();
162 		r = topicMap.updateNode(child, true);
163 		if (r.hasError()) result.addErrorString(r.getErrorString());
164 		r = topicMap.updateNode(parent, true);
165 		if (r.hasError()) result.addErrorString(r.getErrorString());
166 		environment.logDebug("StructuredConversationModel.transcludeChildNodee "+result.getErrorString());
167 		return result;
