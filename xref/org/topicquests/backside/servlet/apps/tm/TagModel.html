42  	 * 
43  	 */
44  	public TagModel(ServletEnvironment env) {
45  		environment = env;
46  		topicMap = environment.getTopicMapEnvironment().getDatabase();
47  		nodeModel = topicMap.getSubjectProxyModel();
48  	}

50  	/* (non-Javadoc)
51  	 * @see org.topicquests.backside.servlet.apps.tm.api.ITagModel#addTagsToNode(org.topicquests.model.api.node.ISubjectProxy, java.lang.String[], org.topicquests.model.api.ITicket)
52  	 */
53  	@Override
54  	public IResult addTagsToNode(ISubjectProxy node, List<String> tagNames,
55  			ITicket credentials) {
56  		String userId = credentials.getUserLocator();
57  		IResult result = new ResultPojo();
58  		IResult r;
59  		int len = tagNames.size();
60  		String name,lox;
61  		ISubjectProxy tag;
62  		for (int i=0;i<len;i++) {
63  			name = tagNames.get(i);
64  			lox = tagNameToLocator(name);
65  			//do we already know this tag?
66  			r = topicMap.getNode(lox, credentials);
67  			if (r.hasError())
68  				result.addErrorString(r.getErrorString());
69  			tag = (ISubjectProxy)r.getResultObject();
70  			if (tag == null) {
71  				//create a tag
72  				environment.logDebug("TagModel.addTagToNode-1 "+lox+" "+userId);
73  				System.out.println("TagModel.addTagToNode-1 "+lox+" "+userId);
74  				tag = nodeModel.newInstanceNode(lox, INodeTypes.TAG_TYPE, name, "", "en", userId, 
75  						ICoreIcons.TAG_SM, ICoreIcons.TAG, false);
76  				r = topicMap.putNode(tag);
77  				if (r.hasError())
78  					result.addErrorString(r.getErrorString());
79  			}
80  			//relate the tag to the user
81  			r = topicMap.getNode(userId, credentials);
82  			if (r.hasError())
83  				result.addErrorString(r.getErrorString());
84  			ISubjectProxy user = (ISubjectProxy)r.getResultObject();
85  			if (user != null) {
86  				environment.logDebug("TagModel.addTagToNode-2 "+tag+" "+user);
87  				System.out.println("TagModel.addTagToNode-2 "+tag+" "+user);
88  				r = nodeModel.relateExistingNodesAsPivots(tag, user, ISocialBookmarkLegend.TAG_USER_RELATION_TYPE, userId, 
89  						ICoreIcons.RELATION_ICON_SM, ICoreIcons.RELATION_ICON, false, false);
90  				if (r.hasError())
91  					result.addErrorString(r.getErrorString());
92  			}
93  			//relate the tag to the topic
94  			environment.logDebug("TagModel.addTagToNode-3 "+tag+" "+node);
95  			System.out.println("TagModel.addTagToNode-3 "+tag+" "+node);
96  			r = nodeModel.relateExistingNodesAsPivots(tag, node, ISocialBookmarkLegend.TAG_BOOKMARK_RELATION_TYPE, userId, 
97  					ICoreIcons.RELATION_ICON_SM, ICoreIcons.RELATION_ICON, false, false);
98  			if (r.hasError())
99  				result.addErrorString(r.getErrorString());
100 			//relating does all the puts
101 		}
102 		return result;
103 	}

110 	private String tagNameToLocator(String name) {
111 		String result = name;
112 		result = result.toLowerCase();
113 		result = result.replaceAll(" ", "_");
114 		result = result.replaceAll("'", "_");
115 		result = result.replaceAll(":", "_");
116 //		result = result.replaceAll("\+", "P");
117 //		result = result.replaceAll("-", "M");
118 		result = result+"_TAG";
119 		return result;
120 	}
