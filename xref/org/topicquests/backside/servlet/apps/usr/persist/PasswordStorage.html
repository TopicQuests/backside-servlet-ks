37  	public static final String PBKDF2_ALGORITHM = "PBKDF2WithHmacSHA1";
38  	// These constants may be changed without breaking existing hashes.
39  	public static final int SALT_BYTE_SIZE = 24;
40  	public static final int HASH_BYTE_SIZE = 18;
41  	public static final int PBKDF2_ITERATIONS = 64000;
42  	// These constants define the encoding and may not be changed.
43  	public static final int HASH_SECTIONS = 5;
44  	public static final int HASH_ALGORITHM_INDEX = 0;
45  	public static final int ITERATION_INDEX = 1;
46  	public static final int HASH_SIZE_INDEX = 2;
47  	public static final int SALT_INDEX = 3;
48  	public static final int PBKDF2_INDEX = 4;
49  
50  	public static String createHash(String password)
51  			throws CannotPerformOperationException {
52  		return createHash(password.toCharArray());
53  	}
54  
55  	public static String createHash(char[] password)
56  			throws CannotPerformOperationException {
57  		// Generate a random salt
58  		SecureRandom random = new SecureRandom();
59  		byte[] salt = new byte[SALT_BYTE_SIZE];
60  		random.nextBytes(salt);
61  
62  		// Hash the password
63  		byte[] hash = pbkdf2(password, salt, PBKDF2_ITERATIONS, HASH_BYTE_SIZE);
64  		int hashSize = hash.length;
65  
66  		// format: algorithm:iterations:hashSize:salt:hash
67  		String parts = "sha1:" + PBKDF2_ITERATIONS + ":" + hashSize + ":" + toBase64(salt) + ":" + toBase64(hash);
68  		return parts;
69  	}

72  			throws CannotPerformOperationException, InvalidHashException {
73  		return verifyPassword(password.toCharArray(), correctHash);
74  	}

76  	public static boolean verifyPassword(char[] password, String correctHash)
77  			throws CannotPerformOperationException, InvalidHashException {
78  		// Decode the hash into its parameters
79  		String[] params = correctHash.split(":");
80  		if (params.length != HASH_SECTIONS) {
81  			throw new InvalidHashException(
82  					"Fields are missing from the password hash."
83  			);
84  		}
85  
86  		if (!params[HASH_ALGORITHM_INDEX].equals("sha1")) {
87  			throw new CannotPerformOperationException("Unsupported hash type.");
88  		}
89  
90  		int iterations = 0;
91  		try {
92  			iterations = Integer.parseInt(params[ITERATION_INDEX]);
93  		} catch (NumberFormatException ex) {
94  			throw new InvalidHashException("Could not parse the iteration count as an integer.", ex);
95  		}
96  
97  		if (iterations < 1) {
98  			throw new InvalidHashException("Invalid number of iterations. Must be >= 1.");
99  		}
100 
101 
102 		byte[] salt = null;
103 		try {
104 			salt = fromBase64(params[SALT_INDEX]);
105 		} catch (IllegalArgumentException ex) {
106 			throw new InvalidHashException("Base64 decoding of salt failed.", ex);
107 		}
108 
109 		byte[] hash = null;
110 		try {
111 			hash = fromBase64(params[PBKDF2_INDEX]);
112 		} catch (IllegalArgumentException ex) {
113 			throw new InvalidHashException("Base64 decoding of pbkdf2 output failed.", ex);
114 		}
115 
116 
117 		int storedHashSize = 0;
118 		try {
119 			storedHashSize = Integer.parseInt(params[HASH_SIZE_INDEX]);
120 		} catch (NumberFormatException ex) {
121 			throw new InvalidHashException("Could not parse the hash size as an integer.", ex);
122 		}
123 
124 		if (storedHashSize != hash.length) {
125 			throw new InvalidHashException("Hash length doesn't match stored hash length.");
126 		}
127 
128 		// Compute the hash of the provided password, using the same salt,
129 		// iteration count, and hash length
130 		byte[] testHash = pbkdf2(password, salt, iterations, hash.length);
131 		// Compare the hashes in constant time. The password is correct if
132 		// both hashes match.
133 		return slowEquals(hash, testHash);
134 	}

136 	private static boolean slowEquals(byte[] a, byte[] b) {
137 		int diff = a.length ^ b.length;
138 		for (int i = 0; i < a.length && i < b.length; i++)
139 			diff |= a[i] ^ b[i];
140 		return diff == 0;
141 	}

143 	private static byte[] pbkdf2(char[] password, byte[] salt, int iterations, int bytes)
144 			throws CannotPerformOperationException {
145 		try {
146 			PBEKeySpec spec = new PBEKeySpec(password, salt, iterations, bytes * 8);
147 			SecretKeyFactory skf = SecretKeyFactory.getInstance(PBKDF2_ALGORITHM);
148 			return skf.generateSecret(spec).getEncoded();
149 		} catch (NoSuchAlgorithmException ex) {
150 			throw new CannotPerformOperationException("Hash algorithm not supported.", ex);
151 		} catch (InvalidKeySpecException ex) {
152 			throw new CannotPerformOperationException("Invalid key spec.", ex);
153 		}
154 	}
