60  	public void handleGet(HttpServletRequest request, HttpServletResponse response, ITicket credentials, JSONObject jsonObject) throws ServletException, IOException {
61  		JSONObject returnMessage = newJSONObject();
62  		String message = "", rtoken = "";
63  		String verb = getVerb(jsonObject);
64  		System.out.println("UserHandler.get verb: " + verb);
65  		int code = 0;
66  		IResult r;
67  		if (verb.equals(IUserMicroformat.LIST_USERS)) {
68  			String startS = getItemFrom(jsonObject);
69  			String countS = getItemCount(jsonObject);
70  			int start = 0, count = -1;
71  			if (!startS.equals("")) {
72  				try {
73  					start = Integer.valueOf(startS);
74  				} catch (Exception e1) {
75  				}
76  			}
77  			if (!countS.equals("")) {
78  				try {
79  					count = Integer.valueOf(countS);
80  				} catch (Exception e2) {
81  				}
82  			}
83  			//TODO: note: we are ignoring any SORT modifiers
84  			//This really returns some live cargo in the form of a list of user objects in JSON format
85  			// We are restricting this to: name, email, avatar, homepage, geolocation, role
86  			r = model.listUsers(start, count);
87  			if (r.hasError()) {
88  				code = BaseHandler.RESPONSE_INTERNAL_SERVER_ERROR;
89  				message = r.getErrorString();
90  			} else {
91  				//Time to take that list apart
92  				System.out.println("UserHandler.ListUsers " + r.getResultObject());
93  				if (r.getResultObject() != null) {
94  					List<ITicket> usrs = (List<ITicket>) r.getResultObject();
95  					Iterator<ITicket> itr = usrs.iterator();
96  					List<JSONObject> jsonUsers = new ArrayList<JSONObject>();
97  					while (itr.hasNext()) {
98  						jsonUsers.add(ticketToUser(itr.next()));
99  					}
100 					returnMessage.put(ICredentialsMicroformat.CARGO, jsonUsers);
101 					code = BaseHandler.RESPONSE_OK;
102 					message = "ok";
103 				} else {
104 					message = "Not found";
105 					code = BaseHandler.RESPONSE_OK;
106 				}
107 			}
108 		} else if (verb.equals(IUserMicroformat.GET_USER_BY_EMAIL)) {
109 			String email = getEmail(jsonObject);
110 			r = model.getTicketByEmail(email);
111 			if (r.getResultObject() != null) {
112 				ITicket t = (ITicket) r.getResultObject();
113 				JSONObject jUser = ticketToUser(t);
114 				returnMessage.put(ICredentialsMicroformat.CARGO, jUser);
115 				code = BaseHandler.RESPONSE_OK;
116 				message = "ok";
117 			} else {
118 				message = "Not found";
119 				code = BaseHandler.RESPONSE_OK;
120 			}
121 		} else if (verb.equals(IUserMicroformat.GET_USER_BY_ID)) {
122 			String id = getUserId(jsonObject);
123 			r = model.getTicketById(id);
124 			if (r.getResultObject() != null) {
125 				ITicket t = (ITicket) r.getResultObject();
126 				JSONObject jUser = ticketToUser(t);
127 				returnMessage.put(ICredentialsMicroformat.CARGO, jUser);
128 				code = BaseHandler.RESPONSE_OK;
129 				message = "ok";
130 			} else {
131 				message = "Not found";
132 				code = BaseHandler.RESPONSE_OK;
133 			}
134 		} else if (verb.equals(IUserMicroformat.GET_USER_BY_HANDLE)) {
135 			String handle = getUserHandle(jsonObject);
136 			r = model.getTicketByHandle(handle);
137 			if (r.getResultObject() != null) {
138 				ITicket t = (ITicket) r.getResultObject();
139 				JSONObject jUser = ticketToUser(t);
140 				returnMessage.put(ICredentialsMicroformat.CARGO, jUser);
141 				code = BaseHandler.RESPONSE_OK;
142 				message = "ok";
143 			} else {
144 				message = "Not found";
145 				code = BaseHandler.RESPONSE_OK;
146 			}
147 		} else {
148 			String x = IErrorMessages.BAD_VERB + "-UserServletGet-" + verb;
149 			environment.logError(x, null);
150 			throw new ServletException(x);
151 		}
152 		returnMessage.put(ICredentialsMicroformat.RESP_TOKEN, rtoken);
153 		returnMessage.put(ICredentialsMicroformat.RESP_MESSAGE, message);
154 		super.sendJSON(returnMessage.toJSONString(), code, response);
155 		returnMessage = null;
156 	}

158 	public void handlePost(HttpServletRequest request, HttpServletResponse response, ITicket credentials, JSONObject jsonObject) throws ServletException, IOException {
159 		JSONObject returnMessage = newJSONObject();
160 		String message = "", rtoken = "";
161 		String verb = getVerb(jsonObject);
162 		int code = 0;
163 		IResult r;
164 		System.out.println("UserHandler.post verb: " + verb);
165 		if (verb.equals(IUserMicroformat.NEW_USER)) {
166 			String email = getEmail(jsonObject);
167 			//TODO sanity check
168 			String userName = getUserHandle(jsonObject);
169 			String userId = jsonObject.getAsString(IUserMicroformat.USER_ID);
170 			String fullName = jsonObject.getAsString(IUserMicroformat.USER_FULLNAME);
171 			String password = jsonObject.getAsString(IUserMicroformat.USER_PWD);
172 			String avatar = notNullString(jsonObject.getAsString(IUserMicroformat.USER_AVATAR));
173 			String homepage = notNullString(jsonObject.getAsString(IUserMicroformat.USER_HOMEPAGE));
174 			System.out.println("NEWUSER " + homepage);
175 			String geolocation = notNullString(jsonObject.getAsString(IUserMicroformat.USER_GEOLOC));
176 			String role = notNullString(jsonObject.getAsString(IUserMicroformat.USER_ROLE));
177 			if (role.equals(""))
178 				role = ISecurity.USER_ROLE; //default role
179 			byte[] foo = BaseEncoding.base64().decode(password);
180 			String creds = new String(foo);
181 			r = model.insertUser(email, userName, userId, creds, fullName, avatar, role, homepage, geolocation, true);
182 			System.out.println("NEWUSER2 " + r.getErrorString());
183 			if (r.hasError()) {
184 				code = BaseHandler.RESPONSE_INTERNAL_SERVER_ERROR;
185 				message = r.getErrorString();
186 			} else {
187 				rtoken = newUUID();
188 				message = "ok";
189 				code = BaseHandler.RESPONSE_OK;
190 			}
191 			returnMessage.put(ICredentialsMicroformat.RESP_TOKEN, rtoken);
192 			returnMessage.put(ICredentialsMicroformat.RESP_MESSAGE, message);
193 		} else if (verb.equals(IAdminMicroformat.UPDATE_USER_PASSWORD)) {
194 			String userId = getUserId(jsonObject);
195 			String password = jsonObject.getAsString(IUserMicroformat.USER_PWD);
196 			byte[] foo = BaseEncoding.base64().decode(password);
197 			String creds = new String(foo);
198 			//	System.out.println("FFFF "+creds);
199 			r = model.changeUserPassword(userId, creds);
200 			if (r.hasError()) {
201 				code = BaseHandler.RESPONSE_INTERNAL_SERVER_ERROR;
202 				message = r.getErrorString();
203 			} else {
204 				message = "ok";
205 				code = BaseHandler.RESPONSE_OK;
206 			}
207 		} else if (verb.equals(IAdminMicroformat.REMOVE_USER)) {
208 			String userId = getUserId(jsonObject);
209 			System.out.println("UserHandler.removeUser " + userId);
210 			r = model.removeUser(userId);
211 			if (r.hasError()) {
212 				code = BaseHandler.RESPONSE_OK;
213 				message = r.getErrorString();
214 			} else {
215 				message = "ok";
216 				code = BaseHandler.RESPONSE_OK;
217 			}
218 		} else if (verb.equals(IAdminMicroformat.UPDATE_USER_DATA)) {
219 			String userId = getUserId(jsonObject);
220 			String key = jsonObject.getAsString(IUserMicroformat.PROP_KEY);
221 			String val = jsonObject.getAsString(IUserMicroformat.PROP_VAL);
222 			r = model.insertUserData(userId, key, val);
223 			if (r.hasError()) {
224 				code = BaseHandler.RESPONSE_OK;
225 				message = r.getErrorString();
226 			} else {
227 				message = "ok";
228 				code = BaseHandler.RESPONSE_OK;
229 			}
230 		} else {
231 			String x = IErrorMessages.BAD_VERB + "-UserServletPost-" + verb;
232 			environment.logError(x, null);
233 			throw new ServletException(x);
234 		}
235 
236 		super.sendJSON(returnMessage.toJSONString(), code, response);
237 		returnMessage = null;
