127 	String getCoreQuery(String verb, String userId, String userIP, String sToken) throws ServletException {
128 		try {
129 			JSONObject json = new JSONObject();
130 			json.put(ICredentialsMicroformat.VERB, verb);
131 			json.put(ICredentialsMicroformat.USER_ID, userId);
132 			json.put(ICredentialsMicroformat.SESSION_TOKEN, sToken);
133 			json.put(ICredentialsMicroformat.USER_IP, userIP);
134 			String result = json.toJSONString();
135 			result = URLEncoder.encode(result, "UTF-8");
136 			json = null;
137 			return result;
138 		} catch (Exception e) {
139 			environment.logError(e.getMessage(), e);
140 			throw new ServletException(e);
141 		}
142 	}
